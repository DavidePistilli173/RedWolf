78 template<rw::io::IsFormattable... Args>
79 void message_base(Level level, std::string_view msg, const Args&... args)
80 {
81 if (level >= level_)
82 {
83 rw::dat::LogMsg message{ level, std::vformat(msg, std::make_format_args(args...)) };
84
85 std::scoped_lock lck{ mtx_ };
86
87 switch (level)
88 {
89 case rw::utils::Logger::Level::trace:
90 std::cout << "\033[0m" << std::format("{}", message) << "\033[0m\n";
91 break;
92 case rw::utils::Logger::Level::info:
93 std::cout << "\033[37m" << std::format("{}", message) << "\033[0m\n";
94 break;
95 case rw::utils::Logger::Level::warning:
96 std::cout << "\033[33m" << std::format("{}", message) << "\033[0m\n";
97 break;
98 case rw::utils::Logger::Level::error:
99 std::cout << "\033[31m" << std::format("{}", message) << "\033[0m\n";
100 break;
101 case rw::utils::Logger::Level::fatal:
102 std::cout << "\033[35m" << std::format("{}", message) << "\033[0m\n";
103 throw std::exception("FATAL ERROR THROWN FROM LOGGER");
104 break;
105 }
106 }
107 }
